## 5.2 `tests/services/test_path_resolver.py`

```python
from __future__ import annotations

from pathlib import Path

import pytest

from services.lightrag_mcp.path_resolver import collect_indexable_files, resolve_input_paths


def test_resolve_input_paths_accepts_workspace_relative_paths(tmp_path: Path) -> None:
    readme = tmp_path / "README.md"
    readme.write_text("hello", encoding="utf-8")

    resolved = resolve_input_paths(tmp_path, ["README.md"])

    assert resolved == [readme.resolve()]


def test_collect_indexable_files_recurses_into_directories(tmp_path: Path) -> None:
    docs = tmp_path / "docs"
    docs.mkdir()
    guide = docs / "guide.md"
    guide.write_text("guide", encoding="utf-8")
    notes = docs / "notes.txt"
    notes.write_text("notes", encoding="utf-8")

    files = collect_indexable_files(tmp_path, ["docs"])

    assert files == [guide.resolve(), notes.resolve()]


def test_resolve_input_paths_rejects_escape_paths(tmp_path: Path) -> None:
    outside = tmp_path.parent / "escape.md"
    outside.write_text("escape", encoding="utf-8")

    with pytest.raises(ValueError):
        resolve_input_paths(tmp_path, [str(outside)])
```

## 5.3 `tests/test_mcp_bridge.py`

这个测试用假 manager 验证桥接层是否真的能把远端 tool 变成 LangChain tool。

```python
from __future__ import annotations

from types import SimpleNamespace

import pytest

from ipilot.mcp import bridge
from ipilot.mcp.models import McpConfig, McpServerConfig, RemoteToolSpec


class FakeManager:
    def __init__(self, config):
        self.config = config

    async def describe_tools(self):
        return [
            RemoteToolSpec(
                server_name="lightrag",
                name="rag_query",
                description="query the knowledge base",
                input_schema={
                    "type": "object",
                    "properties": {
                        "question": {"type": "string"},
                        "mode": {"type": "string"},
                    },
                    "required": ["question"],
                },
            )
        ]

    async def call_tool(self, server_name: str, tool_name: str, arguments):
        assert server_name == "lightrag"
        assert tool_name == "rag_query"
        return SimpleNamespace(content=[SimpleNamespace(text=f"called:{arguments['question']}")])


@pytest.mark.asyncio
async def test_build_mcp_tools_async_creates_langchain_tools(monkeypatch):
    monkeypatch.setattr(bridge, "McpClientManager", FakeManager)

    tools = await bridge.build_mcp_tools_async(
        McpConfig(servers=[McpServerConfig(name="lightrag", url="http://127.0.0.1:8091/mcp")])
    )

    assert [tool.name for tool in tools] == ["mcp_lightrag_rag_query"]
    result = await tools[0].ainvoke({"question": "hello", "mode": "hybrid"})
    assert "called:hello" in result
```

## 5.4 `tests/test_toolset.py`

这个测试用来确认本地工具和 MCP 工具会被拼在一起。

```python
from __future__ import annotations

from ipilot.agent import toolset
from ipilot.config.schema import Config, McpServerConfig


class FakeTool:
    def __init__(self, name: str):
        self.name = name


async def fake_build_mcp_tools_async(_config):
    return [FakeTool("mcp_lightrag_rag_query")]


def test_build_agent_tools_merges_local_and_remote(monkeypatch, tmp_path):
    monkeypatch.setattr(toolset, "build_langchain_tools", lambda workspace: [FakeTool("read_file")])
    monkeypatch.setattr(toolset, "build_mcp_tools_async", fake_build_mcp_tools_async)

    config = Config()
    config.agents.defaults.workspace = str(tmp_path)
    config.mcp.servers = [
        McpServerConfig(name="lightrag", url="http://127.0.0.1:8091/mcp", enabled=True, transport="streamable_http")
    ]

    tools = toolset.build_agent_tools(config)

    assert [tool.name for tool in tools] == ["read_file", "mcp_lightrag_rag_query"]
```

## 5.5 `tests/test_runtime_integration.py`

这个测试只确认 runtime 会把 `build_agent_tools(...)` 的结果交给 `build_agent_graph(...)`。

```python
from __future__ import annotations

from types import SimpleNamespace

from ipilot import runtime
from ipilot.agent import models as agent_models
from ipilot.config.schema import Config, McpServerConfig


class FakeGraph:
    def __init__(self, tools):
        self.tools = tools


def test_build_agent_loop_uses_aggregated_tools(monkeypatch, tmp_path):
    monkeypatch.setattr(runtime, "build_agent_tools", lambda config: [SimpleNamespace(name="mcp_lightrag_rag_query")])

    captured = {}

    def fake_build_agent_graph(*, model, tools, middleware, sqlite_path):
        captured["tools"] = tools
        captured["middleware"] = middleware
        captured["sqlite_path"] = sqlite_path
        return FakeGraph(tools)

    monkeypatch.setattr(runtime, "build_agent_graph", fake_build_agent_graph)
    monkeypatch.setattr(agent_models, "build_chat_model", lambda config: SimpleNamespace(bind_tools=lambda tools: tools))

    config = Config()
    config.agents.defaults.workspace = str(tmp_path)
    config.mcp.servers = [
        McpServerConfig(name="lightrag", url="http://127.0.0.1:8091/mcp", enabled=True, transport="streamable_http")
    ]

    loop = runtime.build_agent_loop(config)

    assert [tool.name for tool in captured["tools"]] == ["mcp_lightrag_rag_query"]
    assert isinstance(loop.graph, FakeGraph)
```

## 5.6 手动联调顺序

### 第一步：启动 server

```bash
uv run python -m services.lightrag_mcp.server
```

你要看到的是：

- 进程正常启动
- 监听 `127.0.0.1:8091`
- 没有先死在 LightRAG 初始化

### 第二步：用 Inspector 看工具

另开一个终端：

```bash
npx -y @modelcontextprotocol/inspector
```

连到：

```text
http://127.0.0.1:8091/mcp
```

你应该能看到：

- `rag_query`
- `rag_index_paths`
- `rag_rebuild`

### 第三步：先索引再查询

先调：

```json
{
  "paths": ["README.md", "docs"]
}
```

再调：

```json
{
  "question": "这个项目的主链是什么？",
  "mode": "hybrid",
  "top_k": 20
}
```

### 第四步：跑 iPilot

把 `~/.ipilot/config.json` 里的 MCP server 配好以后，跑：

```bash
uv run ipilot agent -m "先调用 mcp_lightrag_rag_query，看看这个项目的主链结构。"
```

你要看的不是“模型有没有变聪明”，而是：

- 它有没有看见远端 tool
- 它有没有真的触发 tool call
- 最终回答是不是来自 LightRAG 返回

## 5.7 常见坑

### 坑 1：`path_resolver` 里允许越界路径

后果是知识库会被用户传入的任意文件污染。

### 坑 2：`rag_rebuild` 只做半截

如果你只删文件不重建，测试会越来越难排。

### 坑 3：桥接层把远端结果当原始字符串

最好统一做 JSON 文本化，这样后面的 `ToolNode` 输出更稳定。

### 坑 4：一开始就让 `runtime.py` 进 async 化

第一版不需要。先把同步启动链路跑通。

### 坑 5：先测 agent 再测底层

这是最慢的排错路径。先从路径解析、service、bridge 一层层向上排。
